      { name: "ディフェンダー", min: 3, max: 7, hp: 25 },
      { name: "ストライカー", min: 4, max: 10, hp: 18 },
      { name: "スナイパー", min: 2, max: 8, hp: 15 }
    ];
    for (let i = 0; i < enemyCount; i++) {
      let t = types[Math.floor(Math.random() * types.length)];
      let eHp = Math.floor(t.hp * hpMultiplier * (isElite ? 1.5 : 1));
      enemyList.push({
        id: `E${i}`, x: 4, y: 1 + i, side: "B",
        name: `${isElite ? '【ELITE】' : ''}${t.name}-${i+1}`,
        hp: eHp, maxHp: eHp,
        cardPool: [{ title: "通常攻撃", minDice: t.min, maxDice: t.max, desc: "標準的な敵の攻撃" }]
      });
    }
  }
  return enemyList;
}

function initStage(stageNum) {
  currentStage = stageNum;
  if (currentStage > 50) {

    document.getElementById("home-screen").style.display = "flex"; updateHomeUI(); return;

  let kaitoHp = stageNum === 1 ? maxKaito : Math.min(maxKaito, units.find(u => u.id === "A1")?.hp || maxKaito);
  let rinaHp = stageNum === 1 ? maxRina : Math.min(maxRina, units.find(u => u.id === "A2")?.hp || maxRina);

  units = [
    { id: "A1", x: 1, y: 1, side: "A", name: "Kaito(剣士)", hp: kaitoHp, maxHp: maxKaito, target: null, card: null },

  ];

  let enemies = generateEnemiesForStage(currentStage);

  document.getElementById("log-panel").innerHTML = "<div>-- 作戦開始 --</div>";

function startPlayerTurn() {
  currentCommand = 4;
  document.getElementById("command-val").innerText = `4 / 4`;
  hand = [];
  for(let i=0; i<3; i++) hand.push({ ...playerDeck[Math.floor(Math.random() * playerDeck.length)] });
  runEnemyAI();
  renderHand(); drawMap();
}

function runEnemyAI() {
  let enemies = units.filter(u => u.side === "B");
  let allies = units.filter(u => u.side === "A");
  enemies.forEach(e => {
    if (allies.length === 0) return;
    let targetAlly = allies.reduce((prev, curr) => (prev.hp < curr.hp) ? prev : curr);
    let dist = Math.max(Math.abs(e.x - targetAlly.x), Math.abs(e.y - targetAlly.y));
    if (dist > 1) {
      if (e.x > targetAlly.x) e.x--; else if (e.x < targetAlly.x) e.x++;
      if (e.y > targetAlly.y) e.y--; else if (e.y < targetAlly.y) e.y++;
    }
    e.target = targetAlly.id; e.card = e.cardPool[Math.floor(Math.random() * e.cardPool.length)];
  });
}

function drawMap() {
  const mapDiv = document.getElementById("hex-map"); mapDiv.innerHTML = "";
  const xSpacing = 74; const ySpacing = 70;
  for (let y = 0; y < 5; y++) {
    for (let x = 0; x < mapWidth; x++) {
      let cell = document.createElement("div"); cell.className = "hex-cell";
      let xPos = x * xSpacing; let yPos = y * ySpacing;
      if (x % 2 === 1) yPos += ySpacing / 2;
      cell.style.left = `${xPos}px`; cell.style.top = `${yPos}px`;
      cell.setAttribute("data-coord", `${x},${y}`);

      let inner = document.createElement("div"); inner.className = "hex-inner";
      let u = units.find(v => v.x === x && v.y === y);
      if (u) {
        let badge = document.createElement("div"); badge.className = `unit-badge unit-side-${u.side}`;
        badge.innerText = `${u.name}\nHP: ${u.hp}/${u.maxHp}`;
        inner.appendChild(badge);
        if (selectedCardIndex !== null && u.side === "B") cell.classList.add("valid-target");
      }
      cell.appendChild(inner); cell.onclick = () => handleCellClick(x, y); mapDiv.appendChild(cell);
    }
  }
  setTimeout(drawTargetLines, 50);
}

function renderHand() {
  const container = document.getElementById("hand-container"); container.innerHTML = "";
  hand.forEach((card, idx) => {
    let cardDiv = document.createElement("div"); cardDiv.className = "game-card";
    if (selectedCardIndex === idx) cardDiv.classList.add("active-card");
    cardDiv.style.borderTop = `3px solid ${card.color}`;
    cardDiv.style.animationDelay = `${idx * 0.08}s`;
    cardDiv.onclick = () => {
      if (currentCommand < card.cost) { logMessage("光が足りません！"); return; }
      selectedCardIndex = (selectedCardIndex === idx) ? null : idx; drawMap();
    };
    cardDiv.innerHTML = `<div class="card-cost">${card.cost}</div><div class="card-title" style="color:${card.isFurioso?'#ff1744':'#fff'}">${card.title}</div><div class="card-desc">${card.desc}</div>`;
    container.appendChild(cardDiv);
  });
}

function handleCellClick(x, y) {
  if (selectedCardIndex === null) return;
  let card = hand[selectedCardIndex];
  let userUnit = units.find(v => v.name === card.user);
  let targetUnit = units.find(v => v.x === x && v.y === y && v.side === "B");
  if (targetUnit && Math.max(Math.abs(userUnit.x - x), Math.abs(userUnit.y - y)) <= card.range) {
    userUnit.target = targetUnit.id; userUnit.card = card; targetUnit.target = userUnit.id;
    currentCommand -= card.cost; hand.splice(selectedCardIndex, 1); selectedCardIndex = null;
    document.getElementById("command-val").innerText = `${currentCommand} / 4`;
    renderHand(); drawMap();
  }
}

function drawTargetLines() {
  const svg = document.getElementById("arrow-svg"); svg.innerHTML = "";
  const mapRect = document.getElementById("hex-map").getBoundingClientRect();
  units.forEach(u => {
    if (u.target) {
      let targetUnit = units.find(t => t.id === u.target); if (!targetUnit) return;
      let srcCell = document.querySelector(`[data-coord='${u.x},${u.y}']`);
      let dstCell = document.querySelector(`[data-coord='${targetUnit.x},${targetUnit.y}']`);
      if (!srcCell || !dstCell) return;
      let srcRect = srcCell.getBoundingClientRect(); let dstRect = dstCell.getBoundingClientRect();
      let x1 = (srcRect.left + srcRect.width/2) - mapRect.left; let y1 = (srcRect.top + srcRect.height/2) - mapRect.top;
      let x2 = (dstRect.left + dstRect.width/2) - mapRect.left; let y2 = (dstRect.top + dstRect.height/2) - mapRect.top;
      let cx = (x1 + x2) / 2; let cy = Math.min(y1, y2) - (u.side === "A" ? 30 : -30);
      let defs = svg.querySelector("defs") || document.createElementNS("http://www.w3.org/2000/svg", "defs");
      defs.innerHTML = `<marker id="arrow" viewBox="0 0 10 10" refX="6" refY="5" markerWidth="6" markerHeight="6" orient="auto-start-reverse"><path d="M 0 1 L 10 5 L 0 9 z" fill="${u.side==='A'?'#3182ce':'#ff1744'}"/></marker>`;
      svg.appendChild(defs);
      let path = document.createElementNS("http://www.w3.org/2000/svg", "path");
      path.setAttribute("d", `M ${x1} ${y1} Q ${cx} ${cy} ${x2} ${y2}`);
      path.setAttribute("stroke", u.side === "A" ? "#3182ce" : "#ff1744"); path.setAttribute("stroke-width", "3"); path.setAttribute("fill", "none"); path.setAttribute("marker-end", "url(#arrow)");
      svg.appendChild(path);
    }
  });
}

function startBattlePhase() {
  activeMatches = [];
  units.forEach(u => {
    if (u.side === "A" && u.target) {
      let enemy = units.find(e => e.id === u.target);
      if (enemy && enemy.target === u.id) activeMatches.push({ p: u, e: enemy, type: "clash" });
      else if (enemy) activeMatches.push({ p: u, e: enemy, type: "one-sided" });
    }
  });
  units.forEach(u => {
    if (u.side === "B" && u.target) {
      let targetUnit = units.find(a => a.id === u.target);
      if (targetUnit && !targetUnit.target) activeMatches.push({ p: targetUnit, e: u, type: "one-sided-enemy" });
    }
  });
  if (activeMatches.length === 0) { endTurnAll(); return; }
  showClashWindow();
}

  if (match.p.card && match.p.card.isFurioso) {

    document.getElementById("clash-p-card-title").innerText = "Furioso (特異9連撃)";
    document.getElementById("clash-p-card-desc").innerText = match.p.card.desc;
  } else if (match.type === "clash") {

    document.getElementById("clash-p-card-title").innerText = `${match.p.card.title} [${match.p.card.minDice}-${match.p.card.maxDice}]`;
    document.getElementById("clash-p-card-desc").innerText = match.p.card.desc;
  } else {
    document.getElementById("clash-type-title").innerText = "ONE-SIDED !!";
    document.getElementById("clash-p-card-title").innerText = "(無防備)";
    document.getElementById("clash-p-card-desc").innerText = "";
  }
  document.getElementById("clash-e-card-title").innerText = `${match.e.card.title} [${match.e.card.minDice}-${match.e.card.maxDice}]`;
  document.getElementById("clash-e-card-desc").innerText = match.e.card.desc || "";
  document.getElementById("p-dice").innerText = "-"; document.getElementById("e-dice").innerText = "-";
  document.getElementById("clash-action-btn").style.display = "block";
  document.getElementById("clash-overlay").style.display = "flex";
}

function resolveMatch() {
  let match = activeMatches[0];
  document.getElementById("clash-action-btn").style.display = "none";

  let pDiceEl = document.getElementById("p-dice");
  let eDiceEl = document.getElementById("e-dice");
  let winEl = document.getElementById("clash-window-element");
  let flashEl = document.getElementById("clash-flash-element");
  let cutin = document.getElementById("furioso-cutin");

  if (match.p.card && match.p.card.isFurioso) {
    cutin.classList.add("show");
    logMessage(`「すべての武器を取り出す。これが最後の記憶だ」`, "#ff1744");

    setTimeout(() => {
      cutin.classList.remove("show");
      winEl.classList.add("shake");
      
      let strikeCount = 0;
      
      let furiosoTimer = setInterval(() => {
        strikeCount++;
        let pRoll = Math.floor(Math.random() * (match.p.card.maxDice - match.p.card.minDice + 1)) + match.p.card.minDice;
        let eRoll = Math.floor(Math.random() * (match.e.card.maxDice - match.e.card.minDice + 1)) + match.e.card.minDice;
        
        pDiceEl.innerHTML = `<span style="font-size:12px;color:#888;">#${strikeCount}</span> ${pRoll}`;
        eDiceEl.innerText = eRoll;
        
        flashEl.className = "clash-flash flash-furioso";
        void flashEl.offsetWidth; 
        flashEl.classList.add("flash-furioso");

        if (pRoll >= eRoll) {
          match.e.hp -= pRoll;
          logMessage(` Furioso 連撃 #${strikeCount}: [勝利] ${pRoll} ダメージ！`, "#00e5ff");
        } else {
          match.p.hp -= eRoll;
          logMessage(` Furioso 連撃 #${strikeCount}: [相殺失敗] 反撃で ${eRoll} ダメージを受ける！`, "#ff4d4d");
        }

        if (strikeCount >= 9 || match.e.hp <= 0) {
          clearInterval(furiosoTimer);
          setTimeout(() => {
            winEl.classList.remove("shake");
            document.getElementById("clash-overlay").style.display = "none";
            finalizeMatchResult(match);
          }, 600);
        }
      }, 180);
    }, 1000);

    return;
  }

  let rollInterval = setInterval(() => {
    if (match.type === "clash") {
      pDiceEl.innerText = Math.floor(Math.random() * (match.p.card.maxDice - match.p.card.minDice + 1)) + match.p.card.minDice;
    } else {
      pDiceEl.innerText = "X";
    }
    eDiceEl.innerText = Math.floor(Math.random() * (match.e.card.maxDice - match.e.card.minDice + 1)) + match.e.card.minDice;
  }, 50);

  setTimeout(() => {
    clearInterval(rollInterval);

    let pRoll = match.type === "clash" ? Math.floor(Math.random() * (match.p.card.maxDice - match.p.card.minDice + 1)) + match.p.card.minDice : 0;
    let eRoll = Math.floor(Math.random() * (match.e.card.maxDice - match.e.card.minDice + 1)) + match.e.card.minDice;
    pDiceEl.innerText = match.type === "clash" ? pRoll : "X"; 
    eDiceEl.innerText = eRoll;

    winEl.classList.remove("shake");
    flashEl.className = "clash-flash";

    setTimeout(() => {
      winEl.classList.add("shake");

      if (match.type === "clash") {
        if (pRoll > eRoll) { 
          match.e.hp -= pRoll; 
          flashEl.classList.add("flash-win");
          logMessage(`[勝利] ${match.p.name} が勝利！ ${match.e.name} に ${pRoll} ダメージ。`, "#3182ce"); 
        } else if (eRoll > pRoll) { 
          match.p.hp -= eRoll; 
          flashEl.classList.add("flash-lose");
          logMessage(`[敗北] ${match.p.name} が敗北。 ${eRoll} ダメージ。`, "#e53e3e"); 
        } else {
          logMessage(`[相殺] 互いの攻撃が激突し、霧散した。`);
        }
      } else { 
        match.p.hp -= eRoll; 
        flashEl.classList.add("flash-lose");
        logMessage(`[不意打] ${match.e.name} の攻撃！ ${match.p.name} に ${eRoll} ダメージ。`, "#e53e3e"); 
      }
      
      setTimeout(() => {
        document.getElementById("clash-overlay").style.display = "none";
        winEl.classList.remove("shake");
        finalizeMatchResult(match);
      }, 1000);

    }, 200);
  }, 600);
}

function finalizeMatchResult(match) {
  if (match.e.hp <= 0) {
    units = units.filter(u => u.id !== match.e.id);
    if (match.e.id === "BOSS" && currentStage === 10) {
      let furiosoCard = rewardPool.find(c => c.isFurioso);
      playerDeck.push(JSON.parse(JSON.stringify(furiosoCard)));
      alert(`【第10層ボス撃破報酬】\n黒い沈悶の大技「Furioso」を修得し、デッキに組み込みました！`);
    }
  }
  if (match.p.hp <= 0) units = units.filter(u => u.id !== match.p.id);
  
  activeMatches.shift();
  if (activeMatches.length > 0) showClashWindow(); else endTurnAll();
}

function endTurnAll() {
  if (!units.some(u => u.side === "A")) {
    alert(`作戦終了。第 ${currentStage} 層で全滅しました。`);
    document.getElementById("home-screen").style.display = "flex"; updateHomeUI(); return;
  }
  if (!units.some(u => u.side === "B")) {
    showRewardWindow(); return;
  }
  units.forEach(u => { u.target = null; u.card = null; }); startPlayerTurn();
}
